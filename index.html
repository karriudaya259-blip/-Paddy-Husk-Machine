⚙️ How it works – finite state machine

IDLE → HEATING → PRESSING → PLATE READY → IDLE
  • IDLE : waiting for START button
  • HEATING : red LED on, reads DS18B20 temperature every 100ms
  • PRESSING : at 120°C heater off, servo rotates 90°, holds 3 seconds
  • PLATE READY : servo returns, green LED on, then back to IDLE

STOP and RESET buttons work anytime – safety first. The whole logic runs on a standard Arduino Uno.

🔧 Why Arduino is the right choice

  • Low cost & accessible – official boards start at ₹1,800, clones as low as ₹300. Perfect for students and small workshops.
  • Real‑time control – reads the temperature sensor thousands of times per second, controls servo precisely.
  • Scalable to industrial PLCs – same FSM logic runs on Arduino Opta, used for factory automation.
  • Adaptable to any mould – change the press angle and hold time via variables; the hardware remains the same. Produce plates, vases, pots – just swap the die.
  • Huge ecosystem – DS18B20, servo, LCD, all have ready‑to‑use libraries.

Your simulation proves that even a low‑cost Arduino can manage an industrial heat‑press cycle with ±0.5°C accuracy and millisecond timing.

#include <OneWire.h>
#include <DallasTemperature.h>
#include <Servo.h>

#define ONE_WIRE_BUS A0
#define HEATER_LED_PIN 4
#define READY_LED_PIN 6
#define SERVO_PIN 7
#define START_BTN_PIN 11
#define STOP_BTN_PIN 12
#define RESET_BTN_PIN 13

const float TARGET_TEMP = 120.0;
const int PRESS_ANGLE = 90;
const int RELEASE_ANGLE = 0;
const unsigned long PRESS_HOLD_MS = 3000;

OneWire oneWire(ONE_WIRE_BUS);
DallasTemperature sensors(&oneWire);
Servo pressServo;

enum State { IDLE, HEATING, PRESSING, PLATE_READY };
State currentState = IDLE;
float currentTemp = 0.0;
unsigned long pressStartTime = 0;

void setup() {
  Serial.begin(9600);
  sensors.begin();
  pressServo.attach(SERVO_PIN);
  pressServo.write(RELEASE_ANGLE);
  pinMode(HEATER_LED_PIN, OUTPUT);
  pinMode(READY_LED_PIN, OUTPUT);
  pinMode(START_BTN_PIN, INPUT_PULLUP);
  pinMode(STOP_BTN_PIN, INPUT_PULLUP);
  pinMode(RESET_BTN_PIN, INPUT_PULLUP);
  digitalWrite(HEATER_LED_PIN, LOW);
  digitalWrite(READY_LED_PIN, LOW);
  Serial.println("System Ready. Press START.");
}

void loop() {
  sensors.requestTemperatures();
  currentTemp = sensors.getTempCByIndex(0);

  if (digitalRead(START_BTN_PIN) == LOW && currentState == IDLE) {
    currentState = HEATING;
    digitalWrite(HEATER_LED_PIN, HIGH);
    Serial.println("START pressed. Heating started.");
    delay(200);
  }
  if (digitalRead(STOP_BTN_PIN) == LOW && currentState != IDLE) {
    emergencyStop();
    delay(200);
  }
  if (digitalRead(RESET_BTN_PIN) == LOW) {
    resetSystem();
    delay(200);
  }

  switch (currentState) {
    case HEATING:
      Serial.print("Heating... Temperature: ");
      Serial.print(currentTemp);
      Serial.println(" °C");
      if (currentTemp >= TARGET_TEMP) {
        digitalWrite(HEATER_LED_PIN, LOW);
        currentState = PRESSING;
        pressStartTime = millis();
        pressServo.write(PRESS_ANGLE);
        Serial.println("Target reached. Pressing...");
      }
      break;
    case PRESSING:
      if (millis() - pressStartTime >= PRESS_HOLD_MS) {
        pressServo.write(RELEASE_ANGLE);
        currentState = PLATE_READY;
        digitalWrite(READY_LED_PIN, HIGH);
        Serial.println("Pressing complete. Plate ready!");
      }
      break;
    case PLATE_READY:
      delay(2000);
      digitalWrite(READY_LED_PIN, LOW);
      currentState = IDLE;
      Serial.println("System reset. Ready for next cycle.");
      break;
    default: break;
  }
  delay(100);
}

void emergencyStop() {
  digitalWrite(HEATER_LED_PIN, LOW);
  digitalWrite(READY_LED_PIN, LOW);
  pressServo.write(RELEASE_ANGLE);
  currentState = IDLE;
  Serial.println("EMERGENCY STOP: System halted.");
}

void resetSystem() {
  emergencyStop();
  Serial.println("System reset to IDLE.");
}
